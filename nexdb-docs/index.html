quickstart 
// Add dependency: nexdb-core = "0.1.0"
use nexdb_core::NexDb;

#[tokio::main]
async fn main() -> Result<(), Box> {
    // Open or create database
    let db = NexDb::open("app.nex").await?;
    let users = db.collection("users");

    // Insert JSON document
    let id = users.insert(serde_json::json!({
        "name": "Alice", 
        "age": 30
    })).await?;

    // Query with filter
    let results = users.query()
        .where_clause(r#"{"age": {"$gte": 21}}"#)?
        .limit(10)
        .run().await?;
        
    Ok(())
}
# Install: pip install nexdb
from nexdb import NexDb

# Open database connection
with NexDb('app.nex') as db:
    # Create user collection
    db.create_collection('users')

    # Insert document with auto UUID
    user_id = db.insert_auto_id('users', {
        'name': 'Alice',
        'age': 30
    })

    # Retrieve document by ID
    user = db.get('users', user_id)
    print(f"Found user: {user['name']}")
// Install: npm install nexdb
const { NexDb } = require('nexdb');

async function main() {
  // Initialize and connect
  const db = new NexDb('app.nex');
  await db.connect();

  await db.createCollection('users');

  // Insert JSON document
  const id = await db.insertAutoId('users', {
    name: 'Alice',
    age: 30
  });

  // Get document
  const user = await db.get('users', id);
  console.log(`Found: ${user.name}`);

  await db.close();
}

main().catch(console.error);
Features

Everything a modern database should be

Powerful core capabilities, native library clients, and standalone TCP deployments.

Schemaless Documents

Dynamic JSON document storage supporting nested objects, arrays, and standard datatypes. Built with MongoDB-style collections in mind.

WAL Durability

Append-only Write-Ahead Log (WAL) with snapshot-based checkpointing. Protects your in-memory state against crash corruption.

Multi-Language Support

Interact with your database using highly optimized wrappers for Python (via stdin/stdout subprocess) and Node.js alongside Rust.

Auto B-Tree Indexes

Maintains in-memory B-Tree indexes automatically on nested field paths. Drastically speeds up collection scans from O(n) to O(log n).

TCP Server Mode

Built-in TCP listener running a lean JSON-line protocol. Turn NexDB into a remote server with optional secure JWT-based authorization.

Rich Query Builder

Native Rust QueryBuilder. Provides advanced filtering operators ($eq, $gt, $in, $regex, etc.), multi-field sorting, and pagination.
